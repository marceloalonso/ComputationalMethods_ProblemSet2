function bisection(f, a, b; tol=1e-8, max_iter=500)
    # 1) Verificar se o intervalo inicial [a, b]  vlido
    if f(a) * f(b) > 0
        error("f(a)*f(b) > 0. O intervalo [a, b] no 'bracketa' a raiz.")
    end
    
    # Inicializa o contador de iteraes
    iter = 0
    
    # Para referncia, o comprimento inicial do intervalo
    length_initial = abs(b - a)
    
    # Repetir enquanto o intervalo for maior que tol e no excedermos o max_iter
    while (abs(b - a) > tol) && (iter < max_iter)
        iter += 1
        
        # 2) Em cada iterao, calcula o ponto mdio 
        c = (a + b) / 2
        
        # 3) Verifica se f(c) est suficientemente perto de 0 e o retorna caso sim
        if abs(f(c)) < tol
            return c, iter
        end
        
        # 4) Decide em qual subintervalo a raiz est
        if f(a)*f(c) < 0
            b = c  # raiz em [a, c]
        else
            a = c  # raiz em [c, b]
        end
    end
    
    # Se chegou aqui, pode ser que no tenha convergido ou convergiu "no limite" de iteraes
    if iter == max_iter
        error("O mtodo da bisseo no convergiu em $max_iter iteraes.")
    else
        # Retornamos o ponto mdio como aproximao
        c = (a + b) / 2
        return c, iter
    end
end

c) Now use the secant method. How does it compare?

RESPOSTA DO ITEM C:

No nosso caso, o mtodo da bisseo convergiu em 30 iteraes, enquanto o mtodo da secante levou apenas 9 iteraes. Isso ocorre porque a bisseo possui convergncia linear, reduzindo o tamanho do intervalo pela metade a cada iterao, enquanto o mtodo da secante apresenta convergncia superlinear, atualizando as estimativas com base na inclinao da reta secante entre os dois pontos mais atualizados.

Alm disso, o mtodo da bisseco exige dois pontos, a e b, em que a funo tenha sinais opostos, garantindo a existncia de uma raiz no intervalo. J no mtodo da secante, os dois pontos no precisam ter sinais opostos, mas no podem ser muito prximos, pois isso pode gerar um denominador quase nulo e comprometer a estabilidade do clculo.

function secant(f, a, b; tol=1e-8, max_iter=500)
    # Inicializa o contador de iteraes
    iter = 0
    
    # Avalia a funo nos palpites iniciais
    fa = f(a)
    fb = f(b)
    
    while iter < max_iter
        # Se f(b) j  suficientemente pequeno, retorna b como raiz
        if abs(fb) < tol
            return b, iter
        end
        
        # Evita diviso por zero
        if (fb - fa) == 0
            error("Diviso por zero: f(b) - f(a) == 0")
        end
        
        # Calcula o prximo palpite usando a frmula da secante
        c = b - fb * (b - a) / (fb - fa)
        
        # Verifica se a mudana  menor que a tolerncia
        if abs(c - b) < tol
            return c, iter + 1
        end
        
        # Atualiza os palpites para a prxima iterao
        a, fa = b, fb
        b, fb = c, f(c)
        
        iter += 1
    end
    
    error("O mtodo da secante no convergiu em $max_iter iteraes.")
end

Embora o mtodo de Newton-Raphson possa convergir muito rapidamente com um palpite inicial ideal (por exemplo, 3 iteraes com $x_0=0$), sua eficincia depende fortemente da escolha desse palpite. Alm disso, ao contrrio dos mtodos da bisseo e da secante, o Newton-Raphson exige o clculo explcito da derivada $f'(x)$, o que pode ser um problema caso a derivada no exista ou seja difcil de computar. A convergncia do mtodo tambm est ligada ao comportamento dessa derivada, podendo causar oscilaes ou lentido se for muito pequena ou variar abruptamente na regio escolhida.

function newton_raphson(f, df, x0; tol=1e-8, max_iter=100)
    iter = 0
    x = x0
    while iter < max_iter
        fx = f(x)
        if abs(fx) < tol
            return x, iter
        end
        dfx = df(x)
        if abs(dfx) < tol
            error("Derivada muito prxima de zero; mtodo no pode continuar.")
        end
        x = x - fx/dfx
        iter += 1
    end
    error("Newton-Raphson no convergiu em $max_iter iteraes.")
end

function run_global_method(method_symbol::Symbol; tol=1e-4)
    # Cria o otimizador 1D
    opt = Opt(method_symbol, 1)

    # Define limites [0, 10]
    lower_bounds!(opt, [0.0])
    upper_bounds!(opt, [10.0])

    # Define a funo objetivo
    min_objective!(opt, myfun)

    # Ajusta tolerncias
    ftol_abs!(opt, tol)
    ftol_rel!(opt, tol)
    xtol_abs!(opt, tol)
    xtol_rel!(opt, tol)

    # Limite de avaliaes
    maxeval!(opt, 10000)

    # Chute inicial 
    x0 = [0.0]

    # Executa a otimizao
    (minf, minx_vec, ret) = optimize!(opt, x0)
    return (minx_vec[1], minf, ret)
end

function run_nelder_mead(x0)
    # Mtodo NelderMead
    opt = Opt(:LN_NELDERMEAD, 1)

    # Definindo limites inferiores e superiores: [0, 10]
    lower_bounds!(opt, [0.0])
    upper_bounds!(opt, [10.0])

using JuMP
using Ipopt

function solve_pareto(pesos, preferencias, dotacoes, aversao)
    # Nmero de indivduos (N) e de bens (K)
    N = length(pesos)          
    K = size(preferencias, 2)         

    # Conferncia das dimenses
    @assert size(preferencias, 1) == N "A matriz de preferncias deve ter dimenso NK"
    @assert size(dotacoes, 1) == N "A matriz de dotaes deve ter dimenso NK"
    @assert size(dotacoes, 2) == K "A matriz de dotaes deve ter dimenso NK"
    @assert size(aversao, 1) == N "A matriz de averso deve ter dimenso NK"
    @assert size(aversao, 2) == K "A matriz de averso deve ter dimenso NK"

    # Cria o modelo de otimizao com o solver Ipopt
    model = Model(Ipopt.Optimizer)

    # Declarao das variveis de deciso: x[i,k]  0
    @variable(model, x[1:N, 1:K] >= 0)
    
    # Restries de recursos:
    # para cada bem k, a soma das alocaes no pode ultrapassar a soma das dotaes
    for k in 1:K
        @constraint(model, sum(x[i, k] for i in 1:N) <= sum(dotacoes[i, k] for i in 1:N))
    end

    # Funo objetivo:
    # maximizar  pesos[i] *  [ preferencias[i,k] * (x[i,k])^(1 - aversao[i,k]) / (1 - aversao[i,k]) ]
    @objective(model, Max, sum(
        pesos[i] * sum(
            preferencias[i, k] * (x[i, k]^(1 - aversao[i, k])) / (1 - aversao[i, k])
            for k in 1:K
        )
        for i in 1:N
    ))
    
    # Resolve o problema
    JuMP.optimize!(model)

    # Extrai a soluo tima
    x_opt = [value(x[i, k]) for i in 1:N, k in 1:K]

    return x_opt
end

Interpretao dos Resultados

Os resultados indicam como diferentes distribuies de pesos $\lambda$ afetam a alocao tima dos bens entre os indivduos. Quando um nico indivduo recebe peso total (por exemplo, $\lambda = [1, 0, 0, 0, 0, 0]$), ele concentra praticamente todos os bens, enquanto os demais recebem quantidades mnimas. No caso de pesos distribudos de maneira uniforme ($\lambda = [1/6, 1/6, 1/6, 1/6, 1/6, 1/6]$), a alocao se torna mais equitativa, mas ainda reflete as preferncias e dotaes iniciais dos indivduos. A configurao intermediria com $\lambda = [0.2, 0.2, 0.2, 0.2, 0.1, 0.1]$ leva a uma distribuio levemente desigual, beneficiando os primeiros indivduos com maior dotao. Esses padres evidenciam a influncia dos pesos na determinao das alocaes timas e reforam a sensibilidade da soluo de Pareto funo de bem-estar social adotada.
